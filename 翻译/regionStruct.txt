regionStruct = {
    regionName = string, -- 区域内部名称
    localizedName = string, -- 本地化显示名称
    settlementName = string, -- 区域内主要定居点名称
    legioName = string, -- 军团名称（用于罗马派系）
    regionID = int, -- 区域唯一ID
    roadLevel = int, -- 道路等级（在descr_strat中设定）
    farmingLevel = int, -- 农业等级（在descr_strat中设定）
    famineThreat = int, -- 饥荒威胁值
    harvestSuccess = int, -- 收成成功率
    armyCount = int, -- 区域内军队数量
    settlementCount = int, -- 区域内定居点数量
    watchtowerCount = int, -- 区域内瞭望塔数量
    resourceCount = int, -- 区域内资源数量
    fortCount = int, -- 区域内堡垒数量
    colorRed = int, -- 区域地图颜色红色分量
    colorGreen = int, -- 区域地图颜色绿色分量
    colorBlue = int, -- 区域地图颜色蓝色分量
    isSea = bool, -- 是否为海洋区域
    peninsular = bool, -- 是否为半岛
    landMass = landMass, -- 所属陆地板块
    roadToPort = roadStruct, -- 通往港口的道路
    seaConnectedRegionsCount = int, -- 通过海洋连接的相邻区域数量
    loyaltyFactionID = int, -- 忠诚倾向的派系ID
    seaImportRegionsCount = int, -- 海上进口来源区域数量
    tilesBorderingEdgeOfMapCount = int, -- 与地图边缘接壤的地块数量
    devastatedTilesCount = int, -- 荒废地块数量
    regionCapital = settlementStruct, -- 区域首府定居点
    tileCount = int, -- 区域内总地块数量
    fertileTilesCount = int, -- 肥沃地块数量
    neighbourRegionsNum = int, -- 相邻区域数量
    resourceTypesBitMap = int, -- 资源类型位图
    regionCapitalX = int, -- 首府X坐标
    regionCapitalY = int, -- 首府Y坐标
    portEntranceXCoord = int, -- 港口入口X坐标
    portEntranceYCoord = int, -- 港口入口Y坐标
    faction = factionStruct, -- 当前控制派系
    mercPool = mercPool, -- 雇佣兵池
    rebelType = string, -- 叛军类型
    localizedRebelsName = string, -- 本地化叛军名称
    triumphValue = int, -- 凯旋值（用于罗马凯旋式）
    totalTradeValue = int, -- 总贸易价值
    landingPointsCount = int, -- 登陆点数量
    patrolPointsCount = int, -- 巡逻点数量
    getArmy = getArmy, -- 获取区域内的军队
    getFort = getFort, -- 获取区域内的堡垒
    getWatchtower = getWatchtower, -- 获取区域内的瞭望塔
    getResource = getResource, -- 获取区域内的资源
    getNeighbour = getNeighbour, -- 获取相邻区域
    getHiddenResource = getHiddenResource, -- 获取隐藏资源
    setHiddenResource = setHiddenResource, -- 设置隐藏资源
    getSeaConnectedRegion = getSeaConnectedRegion, -- 获取海洋连接的区域
    getSeaImportRegion = getSeaImportRegion, -- 获取海上进口区域
    getRegionSeaEdge = getRegionSeaEdge, -- 获取区域的海上边缘
    getReligionHistory = getReligionHistory, -- 获取宗教历史数据
    getTileBorderingEdgeOfMap = getTileBorderingEdgeOfMap, -- 获取与地图边缘接壤的地块
    getTile = getTile, -- 获取指定地块
    getFertileTile = getFertileTile, -- 获取肥沃地块
    getDevastatedTile = getDevastatedTile, -- 获取荒废地块
    getHostileArmiesStrength = getHostileArmiesStrength, -- 获取敌对军队强度
    hasResourceType = hasResourceType, -- 检查是否拥有某类资源
    getLandingPoint = getLandingPoint, -- 获取登陆点
    getSettlement = getSettlement, -- 获取区域内的定居点
    getResourcesValue = getResourcesValue, -- 获取资源价值
    hasFaction = hasFaction, -- 检查派系是否存在于此区域
    getEnemySettsToFaction = getEnemySettsToFaction, -- 获取对某派系为敌对的定居点
    getNeutralSettsToFaction = getNeutralSettsToFaction -- 获取对某派系为中立的定居点
}
