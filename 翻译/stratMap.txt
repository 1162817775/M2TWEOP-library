    isInBounds = isInBounds,  -- 检查坐标是否有效(参数:x,y 返回:bool)
    getMovePath = getMovePath,  -- 获取移动路径(参数:fromX,fromY,destX,destY 返回:pathTable)
    rebuildFrontiers = rebuildFrontiers  -- 重建边界显示(无参数)
