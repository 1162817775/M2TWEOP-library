battleStruct = {
    battleState = int, -- 战斗状态
    battleType = int, -- 战斗类型
    isNightBattle = bool, -- 是否为夜战
    battleAiEnabled = bool, -- 是否启用战斗AI
    defenderWithdrawn = bool, -- 防御者是否已撤退
    xCoord = int, -- 战斗发生地X坐标
    yCoord = int, -- 战斗发生地Y坐标
    attackingSide = int, -- 攻击方索引
    defendingSide = int, -- 防御方索引
    residenceOwnerFacId = int, -- 驻地所有者派系ID（如攻城战）
    attackerXCoord = int, -- 攻击方起始X坐标
    attackerYCoord = int, -- 攻击方起始Y坐标
    defenderXCoord = int, -- 防御方起始X坐标
    defenderYCoord = int, -- 防御方起始Y坐标
    paused = bool, -- 是否暂停
    isAutoResolve = bool, -- 是否为自动战斗
    isFortBattle = bool, -- 是否为堡垒战斗
    inBattle = bool, -- 是否正在进行战斗
    finishTimer = int, -- 战斗结束计时器
    battleSpeed = float, -- 战斗速度倍数
    secondsPassed = float, -- 已进行秒数
    tickCount = int, -- 滴答计数
    ticksSinceBattleStart = int, -- 自战斗开始以来的滴答数
    hidingEnabledSet = int, -- 是否启用隐藏功能
    fortInfo = fortBattleInfo, -- 堡垒战斗相关信息
    mapWidth = float, -- 战斗地图宽度
    mapHeight = float, -- 战斗地图高度
    terrainFeatures = terrainFeatures, -- 地形特征
    sidesNum = int, -- 参战方数量
    unitSize = int, -- 单位规模设置
    tideOfBattle = int, -- 战斗趋势（优势方）
    playerArmyNum = int, -- 玩家控制的军队数量
    timeWinningAlliance = int, -- 胜利联盟持续时间
    sides = battleSide[8], -- 参战方数组，最大8个
    factionSide = int[31], -- 派系联盟映射表，-1表示不在战斗中，索引为派系ID+1，最大31
    getPlayerArmy = getPlayerArmy, -- 获取玩家军队对象
    getBattleResidence = getBattleResidence, -- 获取战斗中的驻地（如城镇、城堡）
    getUnitByLabel = getUnitByLabel, -- 通过标签获取战斗单位
    getGroupByLabel = getGroupByLabel, -- 通过标签获取战斗组
    getBattleMapHeight = getBattleMapHeight, -- 获取战斗地图高度
    getBattlefieldEngines = getBattlefieldEngines, -- 获取战场上的攻城器械
    getBattleTile = getBattleTile, -- 获取战斗地图上的地块信息
    getZoneID = getZoneID, -- 获取指定位置的区域ID
    getPosPerimeter = getPosPerimeter, -- 获取位置边界
    isZoneValid = isZoneValid, -- 检查区域是否有效
    isRiverBattle = isRiverBattle -- 是否为河流战斗
}
